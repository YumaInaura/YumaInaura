# js

```js
export function sayHello(user) {
  hello();
}

function hello() {
  console.log('Hello')
}

```

# HTML

```html
<script type="module">

import { sayHello } from 'https://codepen.io/yumainaura/pen/XWrBrbL.js';

sayHello();

</script>
```
