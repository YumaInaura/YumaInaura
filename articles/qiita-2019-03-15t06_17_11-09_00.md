# Script

```py
out = {}

import requests, os

token = input_data['token']
channel = input['channel_id']

file_id = input['permalink'].split("/")[5]

share_api_url ='https://slack.com/api/files.sharedPublicURL?token=' + token + '&file=' + file_id + '&pretty=1'

res = requests.get(share_api_url)
results = res.json()

print(results)

output = [out]
```
