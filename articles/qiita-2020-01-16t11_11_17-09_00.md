```rb
raise StandardError.new('ERROR in class')
# StandardError: ERROR in class

raise StandardError, 'ERROR in raise arg'
# StandardError: ERROR in raise arg

raise StandardError.new('ERROR in class'), 'ERROR in raise arg'
# StandardError: ERROR in raise arg

```
