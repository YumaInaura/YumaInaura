```shell
$ echo '{"alice":"value"}{"bob":"value"}' | jq 'select(has("bob"))'
{
  "bob": "value"
}
```

```shell
$ echo '[{"alice":"value"},{"bob":"value"}]' | jq '.[]' | jq 'select(has("bob"))'
{
  "bob": "value"
}
```
