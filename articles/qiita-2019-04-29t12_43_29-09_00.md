# settings.py
```
DATABASES = {
    'default': {
        'ENGINE': 'django.db.backends.mysql', 
        'NAME': 'some',
        'USER': 'root',
        'PASSWORD': '',
        'HOST': '127.0.0.1',
        'PORT': '10101,
    }
}
```

# docker
```
docker run -e MYSQL_ALLOW_EMPTY_PASSWORD=yes -d -p 10101:3306 -v /tmp/data:/var/lib/mysql --name=mysql-example mysql:5.7
```
# create database
```
docker exec -t mysql-example mysql -e 'CREATE DATABASE some;'
```
