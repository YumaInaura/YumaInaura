# js 

```js
class Button extends React.Component {
  constructor() {
    super();
    this.state = {
      count: 1,
    };
  }

  updateCount() {
    this.setState((prevState, props) => {
      return { count: prevState.count + 1 }
    });
  }

  render() {
    return (<button
              onClick={() => this.updateCount()}
            >
              Clicked {this.state.count} times
            </button>);
  }
}

React.render(<Button />, document.getElementById('app'));
```
