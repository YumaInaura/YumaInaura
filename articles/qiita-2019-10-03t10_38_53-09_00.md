```
$ cat /tmp/.env
MY_ENV_VAR1=abcdef
MY_ENV_VAR2="abcdef"
MY_ENV_VAR3=abc def
```

```
$ docker run --env-file=/tmp/.env alpine sh -c 'echo $MY_ENV_VAR1'
abcdef
$ docker run --env-file=/tmp/.env alpine sh -c 'echo $MY_ENV_VAR2'
"abcdef"
$ docker run --env-file=/tmp/.env alpine sh -c 'echo $MY_ENV_VAR3'
abc def
```
