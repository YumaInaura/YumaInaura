#Ruby / # Rails #FactoryBot example of using create and attributes_for behavior without using association notation


```rb
FactoryBot.define do
  factory :user do
    book do |factory_bot|
      if factory_bot.attribute_lists?
        nil
      else
        create :book
      end
    end
  end
end
```
