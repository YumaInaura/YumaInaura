# when Json array  is empty

```
$ if [ $(echo '[]' | jq '. | length') -eq 0 ]; then echo empty; else echo non empty; fi
empty
```
# when Json array  is not empty

```
$ if [ $(echo '[1]' | jq '. | length') -eq 0 ]; then echo empty; else echo non empty; fi
non empty
```
