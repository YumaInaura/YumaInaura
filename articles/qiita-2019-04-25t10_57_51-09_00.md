```
$ echo '[{"count":1},{"count":3},{"count":2}]' | jq 'sort_by(.count)'
[
  {
    "count": 1
  },
  {
    "count": 2
  },
  {
    "count": 3
  }
]
```


```
$ echo '[{"count":1},{"count":3},{"count":2}]' | jq 'sort_by(.count) | reverse'
[
  {
    "count": 3
  },
  {
    "count": 2
  },
  {
    "count": 1
  }
]
```
