  console.log("validate check");

  if(!nameValidateMeta.valid) {
    console.log("name is invalid")

  }
  if(!emailValidateMeta.valid) {
    console.log("email is invalid")
    return;
  }

  console.log("submit");
  console.log(nameValue.value)
  console.log(emailValue.value)
}
