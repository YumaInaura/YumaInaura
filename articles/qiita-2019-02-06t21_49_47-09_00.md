```
import requests
import os

url = 'https://api.medium.com/v1/me'
token = os.environ['TOKEN']

headers = {
 'Authorization': 'Bearer {}'.format(token),
 'Content-Type': 'application/json',
}

res = requests.get(url, headers=headers)

print(res.json())
```
