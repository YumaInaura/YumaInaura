```rb
# 複数の引数を受け取るメソッド
def foo(a, b)
  p a
  p b
end

# カンマ区切りの文字列を split で配列にする
"x,y".split(",")
# => ["x", "y"]

# 配列をアスタリスク * で 引数として渡す
foo(*"x,y".split(","))
# "x"
# "y"

```
