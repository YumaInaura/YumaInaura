# example

```
$ echo "a" > /tmp/a
$ touch /tmp/b
$ if [ -s /tmp/a ]; then echo not empty; else echo empty; fi
not empty
$ if [ -s /tmp/b ]; then echo not empty; else echo empty; fi
empty
$ if [ -s /tmp/c ]; then echo not empty; else echo empty; fi
empty

```
