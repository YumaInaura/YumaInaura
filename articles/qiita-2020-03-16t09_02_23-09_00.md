```golang
// https://golang.org/pkg/math/rand/

package main

import (
	"fmt"
	"math/rand"
	"time"
)

func main() {
	rand.Seed(time.Now().UnixNano())

	for i := 0; i < 100; i++ {
		fmt.Println(rand.Intn(9))
                // e.g exclude zero
                // fmt.Println(1 + rand.Intn(9-1))
	}

}
```
