```vue
<!-- https://vuejs.org/v2/guide/list.html -->

<script src="https://cdn.jsdelivr.net/npm/vue/dist/vue.js"></script>

<body>

  <div id="example">
    <form action="./">
      <select name="param">
        <option v-for="(value, key) in selectableItems" :value="value">
          {{ key }}
        </option>
        <input type="submit">
      </select>
    </form>
  </div>

  <script>
    new Vue({
      el: '#example',
      data: {
        selectableItems: {
          foo: "foo-value",
          bar: "bar-value"
        }
      }
    })
  </script>
</body>
```
