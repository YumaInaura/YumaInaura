emoji: "🖥"

topics: ["Nuxt"]
published: true
---

```vue
<script lang="ts" setup>
const route = useRoute()

watch(
  () => route.query,
  () => {
    console.log(route['query'])
  }
)

function move() {
  navigateTo({
    name: 'example',
    query: { foo: 'bar' },
  })
}
</script>

<template>
  <button @click="move">move</button>
</template>

```
