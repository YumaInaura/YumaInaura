# Unable to headers with only one line no body CSV file in Ruby 2.5

# Ruby 2.5

```rb
require 'csv'

CSV.parse("A,B,C", headers: true).headers
# => []

CSV.parse("A,B,C\nX,Y,Z", headers: true).headers
# => ["A", "B", "C"]

CSV.parse("A,B,C", headers: true).to_s
# => "\n"
```
