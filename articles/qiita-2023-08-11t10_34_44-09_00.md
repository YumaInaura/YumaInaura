# 数値

```ts
let foo: number

foo = 1

// 型 'string' を型 'number' に割り当てることはできません。
// foo = 'AA'
```

# 数値または文字列

```ts
let bar: number | string

bar = 1
bar = 'AAA'
```

# 独自の型

```ts
type Params = {
  name: string
  age: number
}

let params: Params = {
  name: '',
  age: 0,
}

params.name = 'YYY'
params.age = 20
```
