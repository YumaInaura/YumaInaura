```
import re
some_var = "is"
```

```
re.sub(r"(Alice) (Liddell)","\\1 " + some_var  +  " \\2", 'Alice Liddell')

```

```
re.sub(r"(Alice) (Liddell)","\\1 {is} \\2".format(**{"is":some_var}), 'Alice Liddell')
# 'Alice is Liddell'

```

```
re.sub(r"(Alice) (Liddell)", f"\\1 {some_var} \\2", 'Alice Liddell')
#  'Alice is Liddell'
```
