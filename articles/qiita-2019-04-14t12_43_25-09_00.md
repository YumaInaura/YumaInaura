<h1> Dockerfile </h1>

<pre> <code>FROM ruby 
 
 RUN gem install reverse_markdown redcarpet 
 
</code> </pre>

<h1> EXE </h1>

<pre> <code>$ docker build . -t ruby-gems 
</code> </pre>

<pre> <code>$ docker run -it ruby-gems /bin/bash 
</code> </pre>
