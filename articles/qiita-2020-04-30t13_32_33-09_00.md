```vue

<!-- https://vuejs.org/v2/guide/list.html -->

<script src="https://cdn.jsdelivr.net/npm/vue/dist/vue.js"></script>

<body>
  <ul id="example">
    <li v-for="(key, value) in someItems">
      {{ key }} - {{ value }}
    </li>
  </ul>

  <script>
    new Vue({
      el: '#example',
      data: {
        someItems: {
          message1: 'Foo',
          message2: 'Bar',
        }
      }
    })
  </script>
</body>
```
