package main

	copy_slice_without_pass_argument()

	copy_slice_with_lock()
}

func copy_slice_without_pass_argument() {
	var wg sync.WaitGroup
	var numbers = [100]int{}
	copiedNumbers := []int{}

	for i := range numbers {
		wg.Add(1)
		go func() {
			copiedNumbers = append(copiedNumbers, i)
			wg.Done()
		}()
	}

	wg.Wait()

	fmt.Println(len(copiedNumbers))
	fmt.Println(copiedNumbers)
}

func copy_slice_without_lock() {
	var wg sync.WaitGroup
	var numbers = [100]int{}
	copiedNumbers := []int{}

	for i := range numbers {
		wg.Add(1)
		go func(i int) {
			copiedNumbers = append(copiedNumbers, i)
			wg.Done()
		}(i)
	}

	wg.Wait()

	fmt.Println(len(copiedNumbers))
	fmt.Println(copiedNumbers)
}

func copy_slice_with_lock() {
	var mutex = &sync.Mutex{}
	var wg sync.WaitGroup
	var numbers = [100]int{}
	copiedNumbers := []int{}

	for i := range numbers {
		wg.Add(1)
		go func(i int) {
			mutex.Lock()
			copiedNumbers = append(copiedNumbers, i)
			mutex.Unlock()
			wg.Done()
		}(i)
	}

	wg.Wait()

	fmt.Println(len(copiedNumbers))
	fmt.Println(copiedNumbers)
}
