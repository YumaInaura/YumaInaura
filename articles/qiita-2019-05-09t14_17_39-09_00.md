~/.zshrc

```
function peco-history-selection-keep() {
    BUFFER=$(history -n 1 | tail -r  | awk '!a[$0]++' | peco --layout=bottom-up --query="$BUFFER")

    CURSOR=$#BUFFER
}

zle -N peco-history-selection-keep
bindkey '^R' peco-history-selection-keep


```

`--query="$BUFFER"` がポイント
